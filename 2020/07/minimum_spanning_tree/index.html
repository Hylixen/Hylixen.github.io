Minimum Spanning Tree

  2020-07-31   algorithm   MST  Comments Word Count: 496words Read Count: 3minutes

Definition

Spanning tree is a subgraph(every edge belongs to G) of an undirected and connected graph G = (V,E) that spans G (it includes every vertex of G). And a Minimum Spanning Tree is the spanning tree whose costs is minimum. MST is used algorithms approximating the travelling salesman problem, multi-terminal minimum cut problem, minimum-cost weighted perfect matching, cluster analysis, handwriting recognition, image segemention and so on.

implementations

Prim’s algorithm

steps:

  1. Initialize a tree with a single vertex, chosen arbitrarily from the graph
  2. Grow the tree by the minimum weight edge that connect the tree to vertics not yet in it.
  3. Repeat step 2 until all vertices in the tree.

code:

#include <iostream>
#include <vector>
#include <queue>
#include <utility>
#include <functional>


using namespace std;
typedef pair<long long, int> PII;
const int MAX = 1e4 + 5;
bool marked[MAX];
vector<PII> adj[MAX];


long long prim(int x){
    priority_queue<PII,vector<PII>,greater<PII>> Q;
    PII p;
    int y;
    long long cost = 0;
    Q.push(make_pair(0,x));
    while(!Q.empty()){
        p = Q.top();
        Q.pop();
        x = p.second;
        if(marked[x] == true)
            continue;
        cost += p.first;
        marked[x] = true;
        //push every vertex conneted to x and not in the tree
        for(int i = 0;i<adj[x].size();++i){
            y = adj[x][i].second;
            if(marked[y] == false)
                Q.push(adj[x][i]);
        }
    }
    return cost;
}

int main(){
    int nodes = 0, edges = 0,x = 0, y =0;
    long long cost = 0, weight = 0;
    cin >> nodes >> edges;
    for(int i = 0;i < edges;++i){
        cin >> x >> y >> weight;
        adj[x].push_back(make_pair(weight, y));
        adj[y].push_back(make_pair(weight, x));
    }
    cost = prim(1);
    cout << "minimum cost:" << cost << endl;
    return 0;
}

Kruskal’s algorithm

steps:

  1. initialize a disjoint set data structure with elements pointing to themselves
  2. sort the edges by weight with ascending order
  3. pick edges from the minimum weight and check there’s no loop in the forest.

code:

#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>

using namespace std;
const int MAX = 1e4 + 5;
int id[MAX], edges, nodes;
pair<long long, pair<int,int> >p[MAX];

//init the forest
void initialize(int nodes){
    for (int i = 0;i < nodes+1; ++i)
        id[i]=i;
}

//find root and havle the path
int root(int x){
    while(id[x]!=x){
        id[x]=id[id[x]];
        x=id[x];
    }
    return x;
}

//merge two trees
void union_op(int x,int y){
    id[x] = id[y];
}


long long kruskal(pair<long long, pair<int, int>> p[]){
    int x = 0, y = 0;
    long long total = 0;
    for (int i = 0;i < edges;++i){
        x=root(p[i].second.first);
        y=root(p[i].second.second);
        if(x != y){
            union_op(x,y);
            cout << "union: " << x << "," << y <<endl;
            total += p[i].first;
        }
    }
    return total;
}

int main(){
    int nodes = 0,x = 0, y = 0;
    long long cost = 0, w = 0 ;
    cin >> nodes >> edges;
    initialize(nodes);
    for(int i = 0; i < edges; i++){
        cin >> x >> y >> w;
        p[i] = make_pair(w, make_pair(x, y));
    }
    sort(p, p+edges);
    cost = kruskal(p);
    cout << "cost: " << cost << endl;
    return 0;
}

complexity: O(E log V)
